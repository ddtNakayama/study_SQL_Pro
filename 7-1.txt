# tbl_Aとtbl_Bのレコードに差異があるかチェックする
SELECT
	CASE
		# 結合後のレコード数と結合前のtbl_Aとtbl_Bのレコード数と同じかチェックする
		WHEN COUNT(*) = ( SELECT COUNT(*) FROM tbl_A ) 
			AND COUNT(*) = ( SELECT COUNT(*) FROM tbl_B ) THEN
				'等しい'
		ELSE '異なる'
	END
FROM
	(
		SELECT
			*
		FROM
			tbl_A
	# tbl_Aとtbl_Bを結合する（重複行は除く）
	UNION
		SELECT
			*
		FROM
			tbl_B
	) AS TMP
;
